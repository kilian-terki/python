def est_isocele(a,b,c):
    if(a==b or b==c or c==a):
        return True
    else:
        return False
print (est_isocele(5,5,6))

def lpp(a,b):
    if a < b:
        return a
    else:
        return b

def signe (x):
    if x>0:
        return "Positif"
    elif x==0:
        return "Nulle"
    elif x<0:
        return "Negative"
print (signe(5))
